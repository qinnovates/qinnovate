70/// Algorithm 8 `SamplePolyCBD_Î·(B)` on page 23.
71/// Takes a seed as input and outputs a pseudorandom sample from the distribution `D_ğœ‚(ğ‘…_ğ‘)`. <br>
72/// This function is an optimized version that avoids the `BytesToBits` function (algorithm 4).
73///
74/// Input: byte array `B âˆˆ B^{64Â·Î·}` <br>
75/// Output: array `f âˆˆ Z^{256}_q`
76#[must_use]
77pub(crate) fn sample_poly_cbd(byte_array_b: &[u8]) -> [Z; 256] {
78    let eta = u32::try_from(byte_array_b.len()).unwrap() >> 6;
79    debug_assert_eq!(byte_array_b.len(), 64 * eta as usize, "Alg 8: byte array not 64 * eta");
80    let mut array_f: [Z; 256] = [Z::default(); 256];
81    let mut temp = 0;
82    let mut int_index = 0;
83    let mut bit_index = 0;
84    for byte in byte_array_b {
85        temp |= u32::from(*byte) << bit_index;
86        bit_index += 8;
87        while bit_index >= 2 * (eta as usize) {
88            let tmask_x = temp & ((1 << eta) - 1);
89            let x = count_ones(tmask_x);
90            let tmask_y = (temp >> eta) & ((1 << eta) - 1);
91            let y = count_ones(tmask_y);
92            let (mut xx, mut yy) = (Z::default(), Z::default());
93            xx.set_u16(x);
94            yy.set_u16(y);
95            array_f[int_index] = xx.sub(yy);
96            bit_index -= 2 * (eta as usize);
97            temp >>= 2 * (eta as usize);
98            int_index += 1;
99        }
100    }
101    array_f
102}

105// Count u8 ones in constant time (u32 helps perf)
106#[allow(clippy::cast_possible_truncation)] // return x as u16
107fn count_ones(x: u32) -> u16 {
108    let x = (x & 0x5555_5555) + ((x >> 1) & 0x5555_5555);
109    let x = (x & 0x3333_3333) + ((x >> 2) & 0x3333_3333);
110    let x = (x & 0x0F0F_0F0F) + ((x >> 4) & 0x0F0F_0F0F);
111    x as u16
112}
