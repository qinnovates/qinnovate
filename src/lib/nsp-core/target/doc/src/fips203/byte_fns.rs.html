23/// Algorithm 5 `ByteEncode_d(F)` on page 22.
24/// Encodes an array of `d`-bit integers into a byte array, for `1 ≤ d ≤ 12`.
25/// This is an optimized variant (which does not use individual bit functions).
26///
27/// Input: integer array `F ∈ Z^{256}_m`, where `m = 2^d if d < 12` and `m = q if d = 12` <br>
28/// Output: byte array `B ∈ B^{32·d}`
29pub(crate) fn byte_encode(d: u32, integers_f: &[Z; 256], bytes_b: &mut [u8]) {
30    debug_assert_eq!(bytes_b.len(), 32 * d as usize, "Alg 5: bytes_b len is not 32 * d");
31    debug_assert!(
32        integers_f.iter().all(|f| f.get_u32() <= if d < 12 { 1 << d } else { u32::from(Q) }),
33        "Alg 5: integers_f out of range"
34    );
35
36    // temp acts as a bit buffer that we gradually fill and extract bytes from
37    let mut temp = 0u32;
38    // track how many valid bits are currently in temp
39    let mut bit_index = 0;
40    // track where we're writing in the output byte array
41    let mut byte_index = 0;
42
43    // Process each d-bit integer into bytes
44    for coeff in integers_f {
45        // Mask off any bits above d (safety measure)
46        let coeff = coeff.get_u32() & ((1 << d) - 1);
47
48        // Accumulate bits into temp buffer
49        temp |= coeff << bit_index;
50        bit_index += d as usize;
51
52        // Extract complete bytes (8 bits) whenever possible
53        while bit_index > 7 {
54            // Extract lowest byte from temp
55            bytes_b[byte_index] = temp.to_le_bytes()[0];
56
57            // Remove the extracted byte from temp and update counters
58            temp >>= 8;
59            byte_index += 1;
60            bit_index -= 8;
61        }
62    }
63    debug_assert_eq!(byte_index, bytes_b.len(), "Alg 5: left over bytes_b");
64}

67/// Algorithm 6 `ByteDecode_d(B)` on page 22.
68/// Decodes a byte array into an array of d-bit integers, for 1 ≤ d ≤ 12.
69/// This is an optimized variant (which does not use individual bit functions).
70///
71/// Input: byte array `B ∈ B^{32·d}` <br>
72/// Output: integer array `F ∈ Z^256_m`, where `m = 2^d if d < 12` and `m = q if d = 12`
73pub(crate) fn byte_decode(d: u32, bytes_b: &[u8]) -> Result<[Z; 256], &'static str> {
74    let mut integers_f = [Z::default(); 256];
75    debug_assert_eq!(bytes_b.len(), 32 * d as usize, "Alg 6: bytes len is not 32 * d");
76
77    // temp acts as a bit buffer that we gradually fill and extract d-bit integers from
78    let mut temp = 0u32;
79    // track how many valid bits are currently in temp
80    let mut bit_index = 0;
81    // track where we're writing in the output integer array
82    let mut int_index = 0;
83
84    // Process each byte into d-bit integers
85    for byte in bytes_b {
86        // Accumulate bits into temp buffer
87        temp |= u32::from(*byte) << bit_index;
88        bit_index += 8;
89
90        // Extract complete d-bit integers whenever possible
91        #[allow(clippy::cast_possible_truncation)]
92        while bit_index >= d {
93            // Mask off d bits and convert to Z type
94            let mut z = Z::default();
95            z.set_u16((temp & ((1 << d) - 1)) as u16);
96            integers_f[int_index] = z;
97
98            // Remove the extracted bits from temp and update counters
99            bit_index -= d;
100            temp >>= d;
101            int_index += 1;
102        }
103    }
104
105    debug_assert_eq!(int_index, integers_f.len(), "Alg 6: left over integers");
106
107    // Verify all integers are within valid range for the given bit width
108    let m = if d < 12 { 1 << d } else { u32::from(Q) };
109    ensure!(integers_f.iter().all(|e| e.get_u32() < m), "Alg 6: integers out of range");
110    Ok(integers_f)
111}

129        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(123);
130        //let mut integer_array = [Z::default(); 256];
131        for num_bits in 2..12_u32 {
132            for _i in 0..100 {
133                let num_bytes = 32 * num_bits as usize;
134                let mut bytes2 = vec![0u8; num_bytes];
135                let bytes1: Vec<u8> = (0..num_bytes).map(|_| rng.gen()).collect();
136                let integer_array = byte_decode(num_bits, &bytes1).unwrap();
137                byte_encode(num_bits, &integer_array, &mut bytes2);
138                assert_eq!(bytes1, bytes2);
139            }
140        }
